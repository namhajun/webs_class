01. 변수 :데이터 저장

var x = 100;   //변수 z에 문자 100을 저장
var y = 200;   //변수 z에 문자 200을 저장
var z = "javascript"; //변수 z에 문자 javascript를 저장

document.write("*** 01.변수 ***")
document.write(x);
document.write(y);
document.write(z);
